```rust
use std::collections::HashMap;
use std::sync::Arc;
use crate::handler::SoapHandler;
use crate::fault::SoapFault;
use crate::qname::QName;
use crate::xsd::types::TypeRegistry;
use crate::wsdl::resolver::ResolvedWsdl;

pub struct DispatchEntry {
    pub handler: Arc<dyn SoapHandler>,
    pub auth_required: bool,
    /// QName of the input type in TypeRegistry (for XSD-11 validation)
    pub input_type: Option<QName>,
}

pub struct DispatchTable {
    by_element: HashMap<QName, DispatchEntry>,
    by_action: HashMap<String, DispatchEntry>,
}

#[derive(Debug, thiserror::Error)]
pub enum DispatchError {
    #[error("Operation '{0}' is registered but not found in WSDL")]
    UnknownOperation(String),
    #[error("WSDL operation '{0}' has no registered handler")]
    UnregisteredOperation(String),
}

pub fn build_dispatch_table(
    resolved: &ResolvedWsdl,
    handlers: HashMap<String, Arc<dyn SoapHandler>>,
    auth_bypass: &std::collections::HashSet<String>,  // operation names that skip auth (SEC-06)
) -> Result<DispatchTable, DispatchError> { ... }

pub fn route<'a>(
    table: &'a DispatchTable,

    soap_action: Option<&str>,
) -> Result<&'a DispatchEntry, SoapFault> { ... }
```

For build_dispatch_table: iterate over resolved.definition.services → ports → bindings → binding operations. For each binding operation, find the operation name in the handlers map. If a WSDL operation has no registered handler → DispatchError::UnregisteredOperation. If a registered handler has no WSDL operation → DispatchError::UnknownOperation.

The by_element key is the QName of the input message's first part element (document/literal style). Look this up in the type_registry or definition.messages.
